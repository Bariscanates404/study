``` r
files = googledrive::drive_ls("~/IFTTT/Spotify")
lapply(seq(nrow(files)), 
	function(row) {
		googledrive::drive_download(
			googledrive::as_id(files[row, ]$id)
			, type = "txt"
			, path = glue::glue("/Users/mertnuhoglu/gdrive/IFTTT/Spotify/{files[row, ]$name}.txt")
			, overwrite = T
		)
	}
)
```
